63    /**
64     * Instantiates a new dns node repository.
65     *
66     * @param properties the properties
67     * @param ldapTemplateProvider the ldap template provider
68     * @param dhcpRepository the dhcp repository
69     * @param dnsZoneRepository the dns zone repository
70     */
71    public DnsNodeRepositoryImpl(
72        final DomainControllerProperties properties,
73        final ObjectProvider<LdaptiveTemplate> ldapTemplateProvider,
74        final DhcpRepository dhcpRepository,
75        final DnsZoneRepository dnsZoneRepository) {
76      super(properties, ldapTemplateProvider.getIfAvailable(), dhcpRepository, dnsZoneRepository);
77      this.dnsNodeLdapMapperMap = new ConcurrentHashMap<>();
78      this.dnsNodeLdapMapperProvider = (zoneName, unknownFilter) -> new DnsNodeLdapMapper(
79          getProperties(), zoneName, unknownFilter);
80    }

82    /**
83     * Keep dhcp lease caches up to date.
84     */
85    @Scheduled(fixedDelay = 30000L, initialDelay = 2000)
86    public void keepDhcpLeaseCachesUpToDate() {
87      log.trace("msg=[Keeping dhcp lease cache up to date.]");
88      getDhcpRepository().findActiveByIp();
89      getDhcpRepository().findActiveByHostName();
90    }

92    private LdaptiveEntryMapper<DnsNode> getDnsNodeLdapMapper(
93        final String zoneName,
94        final UnknownFilter unknownFilter) {
95      final UnknownFilter filter = unknownFilter(unknownFilter);
96      final String key = zoneName + ":" + filter.name();
97      return dnsNodeLdapMapperMap.computeIfAbsent(
98          key,
99          k -> dnsNodeLdapMapperProvider.getDnsNodeLdapMapper(zoneName, filter));
100   }

115   @Override
116   public Stream<DnsNode> findAll(
117       final String zoneName,
118       final UnknownFilter unknownFilter,
119       final String query) {
120 
121     final SearchRequest searchRequest = new SearchRequest(
122         getProperties().buildDnsNodeBaseDn(zoneName),
123         new SearchFilter(getProperties().getDnsNodeFindAllFilter()));
124     searchRequest.setSearchScope(getProperties().getDnsNodeFindAllSearchScope());
125     searchRequest.setBinaryAttributes("dnsRecord");
126     if (query == null || query.trim().length() == 0) {
127       return getLdapTemplate().findAll(searchRequest, getDnsNodeLdapMapper(zoneName, unknownFilter))
128           .filter(this::isNonExcludedDnsNode)
129           .map(dnsNode -> insertCorrelationValues(zoneName, dnsNode))
130           .map(dnsNode -> insertDhcpLeases(zoneName, dnsNode));
131     } else {
132       return getLdapTemplate().findAll(searchRequest, getDnsNodeLdapMapper(zoneName, unknownFilter))
133           .filter(this::isNonExcludedDnsNode)
134           .map(dnsNode -> insertCorrelationValues(zoneName, dnsNode))
135           .map(dnsNode -> insertDhcpLeases(zoneName, dnsNode))
136           .filter(dnsNode -> this.isQueryResult(dnsNode, query));
137     }
138   }

140   @Override
141   public boolean exists(
142       final String zoneName,
143       final String nodeName,
144       final UnknownFilter unknownFilter) {
145     return isNonExcludedDnsNode(nodeName)
146         && getDnsZoneRepository().exists(zoneName)
147         && getLdapTemplate().exists(DnsNode.builder().name(nodeName).build(),
148         getDnsNodeLdapMapper(zoneName, unknownFilter));
149   }

151   @Override
152   Optional<DnsNode> findOne(
153       final String zoneName,
154       final String nodeName,
155       final UnknownFilter unknownFilter,
156       final boolean withCorrelationValues,
157       final boolean withDhcpLeases) {
158 
159     final SearchFilter searchFilter = new SearchFilter(getProperties().getDnsNodeFindOneFilter());
160     searchFilter.setParameter(0, nodeName);
161     final SearchRequest searchRequest = new SearchRequest(
162         getProperties().buildDnsNodeBaseDn(zoneName),
163         searchFilter);
164     searchRequest.setSearchScope(getProperties().getDnsNodeFindAllSearchScope());
165     searchRequest.setBinaryAttributes("dnsRecord");
166     return getLdapTemplate().findOne(searchRequest, getDnsNodeLdapMapper(zoneName, unknownFilter))
167         .filter(this::isNonExcludedDnsNode)
168         .map(dnsNode -> withCorrelationValues
169             ? insertCorrelationValues(zoneName, dnsNode)
170             : dnsNode)
171         .map(dnsNode -> withDhcpLeases
172             ? insertDhcpLeases(zoneName, dnsNode)
173             : dnsNode);
174   }

176   @Override
177   public Optional<DnsNode> save(
178       final String zoneName,
179       final DnsNode dnsNode) {
180 
181     if (isExcludedDnsNode(dnsNode)) {
182       throw ServiceException.badRequest(
183           "Node name is not allowed.",
184           "org.bremersee:dc-con-app:8dd7165e-89af-4423-900a-5fc0a71fe7bf");
185     }
186     // Collect deleted records and save existing dns node
187     final Set<DnsRecord> deletedRecords = new LinkedHashSet<>();
188     DnsNode newDnsNode = findOne(zoneName, dnsNode.getName(), ALL, false, false)
189         .map(existingDnsNode -> {
190           for (final DnsRecord existingDnsRecord : existingDnsNode.getRecords()) {
191             if (!dnsNode.getRecords().contains(existingDnsRecord)) {
192               deletedRecords.add(existingDnsRecord);
193             }
194           }
195           if (deletedRecords.size() == existingDnsNode.getRecords().size()) {
196             getLdapTemplate().delete(
197                 existingDnsNode,
198                 getDnsNodeLdapMapper(zoneName, ALL));
199             return DnsNode.builder()
200                 .name(dnsNode.getName())
201                 .build();
202           }
203           return getLdapTemplate().save(dnsNode, getDnsNodeLdapMapper(zoneName, ALL));
204         })
205         .orElseGet(() -> DnsNode.builder()
206             .name(dnsNode.getName())
207             .build());
208 
209     // Collect new records
210     final Set<DnsRecord> newRecords = new LinkedHashSet<>();
211     for (final DnsRecord record : dnsNode.getRecords()) {
212       if (!newDnsNode.getRecords().contains(record)) {
213         newRecords.add(record);
214       }
215     }
216 
217     if (newDnsNode.getRecords().isEmpty() && newRecords.isEmpty()) {
218       // The dns node has no records, it will be deleted
219       if (StringUtils.hasText(newDnsNode.getDistinguishedName())) {
220         getLdapTemplate().delete(dnsNode, getDnsNodeLdapMapper(zoneName, ALL));
221       }
222       newDnsNode = null;
223     } else {
224       // Add new record via cli
225       add(zoneName, dnsNode.getName(), newRecords);
226       // Load dns node from ldap
227       newDnsNode = findOne(zoneName, dnsNode.getName(), ALL, false, false)
228           .orElseThrow(() -> ServiceException.internalServerError(
229               "Saving dns node failed.",
230               "org.bremersee:dc-con-app:7eabb994-f6db-49dc-870b-b4e2dd330a4c"));
231     }
232 
233     // Do A record to PTR record synchronization
234     handlePtrRecords(zoneName, dnsNode.getName(), newRecords, deletedRecords);
235 
236     return Optional.ofNullable(newDnsNode);
237   }

239   /**
240    * Add a dns node.
241    *
242    * @param zoneName the zone name
243    * @param nodeName the node name
244    * @param records the records
245    */
246   void add(
247       final String zoneName,
248       final String nodeName,
249       final Collection<DnsRecord> records) {
250 
251     if (zoneName == null || nodeName == null || records == null || records.isEmpty()) {
252       return;
253     }
254 
255     kinit();
256     for (final DnsRecord record : records) {
257       final List<String> commands = new ArrayList<>();
258       sudo(commands);
259       commands.add(getProperties().getSambaToolBinary());
260       commands.add("dns");
261       commands.add("add");
262       commands.add(getProperties().getNameServerHost());
263       commands.add(zoneName);
264       commands.add(nodeName);
265       commands.add(record.getRecordType());
266       commands.add(record.getRecordValue());
267       auth(commands);
268       CommandExecutor.exec(commands, getProperties().getSambaToolExecDir());
269     }
270   }

272   @Override
273   public boolean delete(final String zoneName, final DnsNode node) {
274     if (isExcludedDnsNode(node)) {
275       throw ServiceException.badRequest(
276           "Node name is not allowed.",
277           "org.bremersee:dc-con-app:3e377240-eafd-45ea-9ee6-048ab3ca8cec");
278     }
279     getLdapTemplate().delete(node, getDnsNodeLdapMapper(zoneName, ALL));
280     handlePtrRecords(zoneName, node.getName(), Collections.emptySet(), node.getRecords());
281     return true;
282   }
