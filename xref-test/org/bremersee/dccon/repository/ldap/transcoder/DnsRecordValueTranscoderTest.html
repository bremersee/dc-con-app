62    /**
63     * Decode binary value record a.
64     */
65    @Test
66    public void decodeBinaryValueRecordA() {
67      DnsRecord actual = transcoder.decodeBinaryValue(Base64.getDecoder().decode(A_RECORD_BASE64));
68      assertNotNull(actual);
69      assertEquals(A_RECORD_TYPE, actual.getRecordType());
70      assertEquals(A_RECORD_VALUE, actual.getRecordValue());
71      assertEquals(A_RECORD_VERSION, actual.getVersion());
72      assertEquals(A_RECORD_SERIAL, actual.getSerial());
73      assertEquals(A_RECORD_TTL, actual.getTtlSeconds());
74      assertEquals(A_RECORD_TIME_STAMP, actual.getTimeStamp());
75    }
