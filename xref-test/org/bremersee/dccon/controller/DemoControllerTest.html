20  @SpringBootTest(webEnvironment = SpringBootTest.WebEnvironment.RANDOM_PORT, properties = {
21      "bremersee.security.authentication.enable-jwt-support=false"
22  })
23  @ActiveProfiles({"basic-auth"})
24  @TestInstance(Lifecycle.PER_CLASS) // allows us to use @BeforeAll with a non-static method
25  @TestMethodOrder(MethodOrderer.OrderAnnotation.class)
26  class DemoControllerTest {
27  
28    private static final String user = "admin";
29  
30    private static final String pass = "admin";
31  
32    @Autowired
33    TestRestTemplate restTemplate;
34  
35    @Order(1)
36    @Test
37    void resetData() {
38      ResponseEntity<Void> response = restTemplate
39          .withBasicAuth(user, pass)
40          .postForEntity("/api/reset", null, Void.class);
41      assertEquals(HttpStatus.OK, response.getStatusCode());
42    }
43  
44    @Order(2)
45    @Test
46    void resetDataAndExpect() {
47      ResponseEntity<Void> response = restTemplate
48          .postForEntity("/api/reset", null, Void.class);
49      assertEquals(HttpStatus.UNAUTHORIZED, response.getStatusCode());
50    }
51  
52    @Order(3)
53    @Test
54    void throwError() {
55      ResponseEntity<RestApiException> response = restTemplate
56          .withBasicAuth(user, pass)
57          .getForEntity("/api/error", RestApiException.class);
58      assertEquals(HttpStatus.I_AM_A_TEAPOT, response.getStatusCode());
59      RestApiException actual = response.getBody();
60      assertNotNull(actual);
61      assertEquals("teapot", actual.getMessage());
62      assertNotNull(actual.getCause());
63      assertEquals("Example error message", actual.getCause().getMessage());
64    }

66  }
