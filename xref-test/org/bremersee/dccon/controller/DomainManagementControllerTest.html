20  class DomainManagementControllerTest {
21  
22    private static final String user = "admin";
23  
24    private static final String pass = "admin";
25  
26    @Autowired
27    TestRestTemplate restTemplate;
28  
29    @Test
30    void getPasswordInformation() {
31      ResponseEntity<PasswordInformation> response = restTemplate
32          .withBasicAuth(user, pass)
33          .getForEntity("/api/domain/password-information", PasswordInformation.class);
34      assertEquals(HttpStatus.OK, response.getStatusCode());
35      PasswordInformation actual = response.getBody();
36      assertNotNull(actual);
37      assertNotNull(actual.getMinimumPasswordLength());
38    }
39  
40    @Test
41    void getRandomPassword() {
42      ResponseEntity<Password> response = restTemplate
43          .withBasicAuth(user, pass)
44          .getForEntity("/api/domain/random-password", Password.class);
45      assertEquals(HttpStatus.OK, response.getStatusCode());
46      Password actual = response.getBody();
47      assertNotNull(actual);
48      assertNotNull(actual.getValue());
49    }
